```bash

#!/usr/bin/env python3
"""Validate that files contain no emoji."""
import re
import sys
from pathlib import Path

EMOJI_PATTERN = re.compile(
    "["
    "\U0001F600-\U0001F64F"
    "\U0001F300-\U0001F5FF"
    "\U0001F680-\U0001F6FF"
    "\U0001F1E0-\U0001F1FF"
    "\U00002702-\U000027B0"
    "\U000024C2-\U0001F251"
    "\U0001F900-\U0001F9FF"
    "\U0001F018-\U0001F270"
    "]+",
    flags=re.UNICODE
)

TEXT_EMOJI = ['‚≠ê', '‚úÖ', '‚ùå', '‚ö†Ô∏è', 'üî¥', 'üü°', 'üü¢', '‚ÑπÔ∏è', 'üí°', 'üö®']

files_with_emoji = []
for file_path in sys.argv[1:]:
    try:
        content = Path(file_path).read_text()
        if EMOJI_PATTERN.search(content) or any(e in content for e in TEXT_EMOJI):
            files_with_emoji.append(file_path)
    except:
        pass

if files_with_emoji:
    print("ERROR: Emoji found in files:")
    for f in files_with_emoji:
        print(f"  - {f}")
    print("\nRemove emoji using: python3 scripts/remove-all-emoji.py")
    sys.exit(1)

print("[OK] No emoji found")
EOF

chmod +x scripts/validate-no-emoji.py
```
